How to Change Theme Color ?

                      
  // ----------------------------------------------------
  // File: package/dist/js/app.init.js
  // ----------------------------------------------------    

  $(function () {
    "use strict";
    $("#main-wrapper").AdminSettings({
      // Theme Option: blue_theme | blue_theme | cyan_theme | green_theme | orange_theme | purple_theme 
      ThemeBg: "aqua_theme", // change is here
      SidebarType: "full", 
    });
  });

How to Set Dark Theme ?

                      
// ----------------------------------------------------
// File: Your HTML File
// ----------------------------------------------------

<!-- 1. import style-dark.min.css into the head -->
<link  id="themeColors"  rel="stylesheet" href="../../dist/css/style-dark.min.css" />

<!-- 2. dark logo image -->
<img src="../../dist/images/logos/light-logo.svg" class="light-logo" width="180" alt="" style="">

<nav class="navbar navbar-expand-lg navbar-dark"></nav> <!--// 3. change class navbar-light to navbar-dark -->
